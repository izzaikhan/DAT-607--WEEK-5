games <- players %>%
  pivot_longer(
    cols = matches("^r\\d"),
    names_to = "round",
    values_to = "result_raw"
  ) %>%
  mutate(
    result = str_extract(result_raw, "^[WDL]"),
    opp_num = as.integer(str_extract(result_raw, "\\d+")),
    actual_score = case_when(
      result == "W" ~ 1,
      result == "D" ~ 0.5,
      result == "L" ~ 0,
      TRUE ~ NA_real_
    )
  ) %>%
  filter(!is.na(opp_num), !is.na(actual_score)) %>%
  select(
    player_num = pair_num,
    player_name,
    player_rating,
    round,
    opp_num,
    actual_score
  )

# ℹ 1 more variable: opp_rating <int>
expected_score <- function(r_player, r_opp) {
  1 / (1 + 10 ^ ((r_opp - r_player) / 400))
}

games <- games %>%
  mutate(expected = expected_score(player_rating, opp_rating))

summary_tbl <- games %>%
  group_by(player_num, player_name, player_rating) %>%
  summarise(
    expected_total = sum(expected, na.rm = TRUE),
    actual_total = sum(actual_score, na.rm = TRUE),
    diff = actual_total - expected_total,
    .groups = "drop"
  )

top_over <- summary_tbl %>% arrange(desc(diff)) %>% slice_head(n = 5)
top_under <- summary_tbl %>% arrange(diff) %>% slice_head(n = 5)
